<table>
  <thead>
    <tr>
      <th>CONTRIBUTOR</th>

      <th>DATE (YYYY/MM/DD)</th>
      <th>RESOURCE LINK (PR, Blog, Video)</th>

  </thead>
  <tbody>

      return (
        <tr key={item.username + item.name + index}>
          <td>
            <a href={`https://github.com/${item.username}`}>{item.name}</a>
          </td>
          <td>{item.contribution}</td>
          <td>{item.date}</td>
          <td>
            <a href={item.resource.url}>

        </tr>
      );
    })}
  </tbody>
</table>

export default ({ children }) => {
  data.sort((a, b) => new Date(b.date) - new Date(a.date));
  return (
    <DocsLayout
      title="LinkFree Contributors Changelog"
      section="Getting started"
      name="Community Resources"
    >
      {children}
    </DocsLayout>
  );
};
